
# python calc
num1 = float(input("First Number: "))
func = input(" function(+,-,*,/): ")
num2 = float(input("Second Number: "))
if func == "+":
    sol1 = str(num1 + num2)
    print("Anwser: " + sol1)
if func == "-":
    sol1 = str(num1 - num2)
    print("Anwser: " + sol1)
if func == "*":
    sol1 = str(num1 * num2)
    print("Anwser: " + sol1)
if func == "/":
    sol1 = str(num1 / num2)
    print("Anwser: " + sol1)
